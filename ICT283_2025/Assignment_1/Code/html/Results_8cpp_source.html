258bool Results::ValidateMonth(const SensorLog &sensor_data, const int month,
259 const int year, const string sensor_type) const {
260
261 // Check which sensor_type was given in the parameter
262 // S for Speed
263 // T for Temperature
264 // SR for Solar Radiation
265
266 if (sensor_type == "S") {
267 // Loop through all sensor records and return true if a speed record exists
268 // for the year and month that was given in the parameter
269 for (int index = 0; index < sensor_data.size(); index++) {
270 if (sensor_data[index].date.GetMonth() == month &&
271 sensor_data[index].date.GetYear() == year &&
272 sensor_data[index].speed != 0) {
273 return true;
274 }
275 }
276 }
277 if (sensor_type == "T") {
278 // Loop through all sensor records and return true if a temperature record
279 // exists for the year and month that was given in the parameter
280 for (int index = 0; index < sensor_data.size(); index++) {
281 if (sensor_data[index].date.GetMonth() == month &&
282 sensor_data[index].date.GetYear() == year &&
283 sensor_data[index].temperature != 0) {
284 return true;
285 }
286 }
287 }
288 if (sensor_type == "SR") {
289 // Loop through all sensor records and return true if a Solar Radiation
290 // record exists for the year and month that was given in the parameter
291 for (int index = 0; index < sensor_data.size(); index++) {
292 if (sensor_data[index].date.GetMonth() == month &&
293 sensor_data[index].date.GetYear() == year &&
294 sensor_data[index].solar_radiation != 0) {
295 return true;
296 }
297 }
298 }
299 // Else return false
300 return false;
301}
