300bool Results::ValidateMonth(const SensorLog &sensor_data, const int month,
301 const int year, const string sensor_type) const {
302
303 // Check which sensor_type was given in the parameter
304 // S for Speed

322 for (int index = 0; index < sensor_data.size(); index++) {

324 sensor_data[index].date.GetYear() == year &&
325 sensor_data[index].temperature != 0) {

333 for (int index = 0; index < sensor_data.size(); index++) {
